-- Create Patients Table
CREATE TABLE patients (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    name TEXT NOT NULL,
    aadhar_no TEXT UNIQUE NOT NULL,
    mobile_no TEXT NOT NULL,
    state TEXT,
    city TEXT,
    district TEXT,
    password TEXT NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT TIMEZONE('utc'::text, NOW())
);

-- Create Doctors Table
CREATE TABLE doctors (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    name TEXT NOT NULL,
    aadhar_no TEXT UNIQUE NOT NULL,
    mobile_no TEXT NOT NULL,
    state TEXT,
    city TEXT,
    district TEXT,
    password TEXT NOT NULL,
    hospital_name TEXT NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT TIMEZONE('utc'::text, NOW())
);

-- Create Medical History Table
CREATE TABLE medical_history (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    patient_name TEXT,
    patient_aadhar_no TEXT REFERENCES patients(aadhar_no) ON DELETE CASCADE,
    doctor_aadhar_no TEXT REFERENCES doctors(aadhar_no) ON DELETE SET NULL,
    past_medical_record TEXT NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT TIMEZONE('utc'::text, NOW())
);
